  int NthTermOfSeries(int n)
    {
       int i=0;
       int ans=0; 
       while (n)
       {
           if(n&1)
           {
               ans=ans+4*pow(10,i);
               
               n=(n-1)/2;
               
           }else{
               
               ans=ans+7*pow(10,i);
               
               n=(n-2)/2;
               
           }
           
           i++;
       }
       
       
       
       
       return ans;
    }
